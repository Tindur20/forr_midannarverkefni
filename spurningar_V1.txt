   def move(self, dx=0, dy=0):
        if not self.collide_with_walls(dx, dy):		# hér er það sem leifir ekki playerinum að hreyfa sig ef hann er að reyna að labba í gegnum veg.
            self.x += dx
            self.y += dy

    def collide_with_walls(self, dx=0, dy=0):
        for wall in self.game.walls:
            if wall.x == self.x + dx and wall.y == self.y + dy: # checkar hvort player er að fara að inn í vegginn í bæði x og y
                return True					# returnar true ef hann er að lenda á veg og er að reyna að komast í gegnum hann
        return False						# returnar false ef hann er ekki að fara lenda á vegg
